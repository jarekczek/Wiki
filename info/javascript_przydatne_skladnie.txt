    <p>
      <!-- <a id="pdflink">Pasek PDF</a> -->
      <a id="pdflink">link</a>
    </p>
    <script>
      //https://blog.jayway.com/2017/07/13/open-pdf-downloaded-api-javascript/
      var linkElem = document.querySelector('#pdflink')
      var filename = "pasek.txt";
      var forceDownload = ( filename != null );
      if (forceDownload)
        linkElem.download = filename;
      var blob = new Blob(["napis"], {type: "text/plain"});
      if (window.navigator && window.navigator.msSaveOrOpenBlob) {
        // IE.
        //https://stackoverflow.com/a/24354303/772981
        //linkElem.download = filename;
        linkElem.href = '#';
        linkElem.onclick = function() {
          if (forceDownload)
            window.navigator.msSaveOrOpenBlob(blob, filename);
          else
            window.navigator.msSaveOrOpenBlob(blob);
        }
      } else {
        // Firefox, Chrome.
        var objectUrl = window.URL.createObjectURL(blob);
        document.querySelector('#pdflink').href = objectUrl;
      }
    </script>
